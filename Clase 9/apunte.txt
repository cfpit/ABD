Funciones de tabla en linea
CREATE FUNCTION nombre(@variable)
returns table
as
	return (SELECT		campos
			FROM		tabla1 as t1
			INNER JOIN	tabla2 as t2
			ON			t1.id=t2.id
			WHERE		campo = @variable)
